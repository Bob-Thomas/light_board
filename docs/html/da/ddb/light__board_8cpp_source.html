15 void light_board::update() {
16  uint32_t distance = (uint32_t) height_sensor.get_distance();
17  if (!peak_reached && current_jump_height < (int) distance) {
18  current_jump_height = distance;
19  } else if (!peak_reached && jump_started && current_jump_height > (int) distance) {
20  peak_reached = true;
21  jump_time = hwlib::now_us();
22  }
23  if (!jump_started && (int) distance >= min_jump_height + 10) {
24  jump_started = true;
25  }
26  if (jump_started && (int) distance <= min_jump_height + 3) {
27  jump_time = (hwlib::now_us() - jump_time) - DISTANCE_CALIBRATION;
28  if (current_jump_height < 200) {
29  highscores.add_score(current_jump_height);
30  last_jump = current_jump_height;
31  }
32  jump_started = false;
33  peak_reached = false;
34  current_jump_height = 0;
35  }
36  if (distance > 200) {
37  idle_lights();
38  } else {
39  if (jump_started) {
40  jump_lights(distance);
41  } else {
42  riding_lights();
43  }
44  }
45 }

75 void light_board::riding_lights() {
76  if (last_jump == highscores.get_highscores()[0]) {
77  for (uint32_t j = 0; j < strip.getNumLeds(); j++) {
78  strip.set_pixel_color(j, get_score_color(last_jump));
79  strip.update();
80  hwlib::wait_ms(10);
81  }
82  for (int i = 0; i < (int) strip.getNumLeds(); i++) {
83  byte red = sin(6 * i + 0) * 127 + 128;
84  byte green = sin(6 * i + 2) * 127 + 128;
85  byte blue = sin(6 * i + 4) * 127 + 128;
86  strip.set_pixel_color(i, red, green, blue);
87  strip.update();
88  hwlib::wait_ms(10);

91  strip.set_pixel_color(j, get_score_color(last_jump));

103 void light_board::jump_lights(int distance) {
104  if (distance > max_jump_height) {
105  distance = (uint32_t) max_jump_height;
106  }
107  if (distance < min_jump_height) {
108  distance = (uint32_t) min_jump_height;
109  }
110  int height_percentage = ((distance - min_jump_height) * 100) / (max_jump_height - min_jump_height);
111  int color = 255 / 100 * height_percentage;
112  if (distance <= min_jump_height) {
113  color = 0;
114  }
115  for (uint32_t j = 0; j < strip.getNumLeds(); j++) {
116  strip.set_pixel_color(j, color, 0, 0);
117  }
118  strip.update();
119 }
