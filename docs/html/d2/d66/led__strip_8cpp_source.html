47 void led_strip::update() {
48  for (int i = 0; i < NUM_LEDS; i++) {
49  Color color = colors[i];
50  for (int j = 0; j < COLOR_CHANNELS; j++) {
51  byte c = color.getColor()[j];
52  for (int bitNum = CHANNEL_BIT - 1; bitNum >= 0; bitNum--) {
53  clock.set(0);
54  byte mask = 1 << bitNum;
55  data.set(c & mask);
56  // Maximum input clock frequency for the WS2801 is 25MHz,
57  // so no delay is required with a 16MHz Arduino Uno.
58  clock.set(1);
59  }
60  }
61  }
62  clock.set(0);
63  hwlib::wait_ms(1);
64 }
